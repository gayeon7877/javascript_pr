switch()문은 다중 if문과 매우 비슷하다. 

        switch(조건비교할 변수){
            case 비교값1:
                실행할 코드;
                break;

            case 비교값2:
                실행할 코드;
                break;

            case 비교값3:
                실행할 코드;
                break;

            default:
                실행할 코드;  
        }
